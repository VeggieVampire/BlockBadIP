# BlockBadIP
Simple iptable blocking script. IP list from blocklist.de

# Install
sudo apt-get install -y git <br>
cd ~ <br>
git clone https://github.com/VeggieVampire/BlockBadIPs <br>
cd BlockBadIPs<br>
chmod 777 * <br>

# Run
cd BlockBadIPs <br>
./BlockBadIPs.sh <br>

# UnInstall
cd BlockBadIPs <br>
./UnBlockAllBadIPs.sh <br>
./UnBlockAllBadIPs.sh <br>
cd ..
rm -rf BlockBadIPs
